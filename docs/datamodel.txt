PREFIX rdfs: <http://www.w3.org/2000/01/rdf-schema#> 
PREFIX owl: <http://www.w3.org/2002/07/owl#> 
PREFIX xsd: <http://www.w3.org/2001/XMLSchema#> 
PREFIX rdf: <http://www.w3.org/1999/02/22-rdf-syntax-ns#> 
PREFIX dc: <http://purl.org/dc/terms/> 
PREFIX bibo: <http://purl.org/ontology/bibo/> 
PREFIX geo: <http://opengis.net/ont/geosparql>

SELECT ?spatial object ?document

WHERE {
	spatial object DC:isReferencedBy document .
	document dc:references spatial object .
	spatial object geo:hasGeometry geo:Geometry .
	document bibo:producer dc:Agent .
}
